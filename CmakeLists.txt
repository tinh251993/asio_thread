cmake_minimum_required(VERSION 3.12)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=gnu++11")

project(aiso_boost)

find_package(Boost 1.55.0 REQUIRED COMPONENTS system filesystem)

add_executable(
	btvn
	main.cpp
	Generator.hpp
	Generator.cpp
)

include_directories(cilent ${Boost_INCLUDE_DIRS})
link_directories(cilent ${Boost_LIBRARY_DIRS})

target_link_libraries(cilent ${Boost_LIBRARIES})
target_link_libraries(cilent ws2_32 wsock32 )
